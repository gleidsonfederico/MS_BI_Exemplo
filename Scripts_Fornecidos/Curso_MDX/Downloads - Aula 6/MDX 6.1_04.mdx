WITH MEMBER [Measures].[Faturamento de Vendas]
AS
'[Measures].[Faturamento]', FORMAT_STRING = "#,###.00"
SELECT 
ORDER ( 
BottomCount  (
{[Cliente].[Hierarquia Geográfica].[Nível Cidade].Members} , 3, [Measures].[Faturamento]
)
, [Measures].[Faturamento], BDESC)
on rows,
([Tempo].[Hierarquia do Mês e Ano].[Ano].&[2014]) on columns
from [COMPLETO]
WHERE [Measures].[Faturamento de Vendas]
