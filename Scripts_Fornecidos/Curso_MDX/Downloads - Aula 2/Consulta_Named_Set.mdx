WITH SET [SEIS PRIMEIROS MESES 2013] AS 
	'{[TEMPO].[MES E ANO].&[2013]&[01-2013]&[01-2013]&[01-2013],
	[TEMPO].[MES E ANO].&[2013]&[01-2013]&[01-2013]&[02-2013],
	[TEMPO].[MES E ANO].&[2013]&[01-2013]&[01-2013]&[03-2013],
	[TEMPO].[MES E ANO].&[2013]&[01-2013]&[02-2013]&[04-2013],
	[TEMPO].[MES E ANO].&[2013]&[01-2013]&[02-2013]&[05-2013],
	[TEMPO].[MES E ANO].&[2013]&[01-2013]&[02-2013]&[06-2013]}'
	MEMBER [TEMPO].[MES E ANO].[SOMA DOS SEIS PRIMEIROS MESES DE 2013] AS 
	'SUM([SEIS PRIMEIROS MESES 2013],[MEASURES].[FATURAMENTO LIQUIDO])'
SELECT	
{
[SEIS PRIMEIROS MESES 2013],
[SOMA DOS SEIS PRIMEIROS MESES DE 2013]
}
	ON ROWS,
{
[MEASURES].[FATURAMENTO LIQUIDO]
}		
		ON COLUMNS
FROM	[COMPLETO]
