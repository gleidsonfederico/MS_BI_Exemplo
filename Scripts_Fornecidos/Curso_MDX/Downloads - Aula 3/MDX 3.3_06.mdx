CROSSJOIN (
({[Cliente].[Cidade].&[Rio de Janeiro], [Cliente].[Cidade].&[São Paulo], [Cliente].[Cidade].&[Belo Horizonte]}),
({[Produto].[Marca].&[Festival de Sabores], [Produto].[Marca].&[Pedaços de Frutas], [Produto].[Marca].&[Sabor da Montanha]}) 
)
,[Measures].[Faturamento], BDESC
)
ON ROWS,
({ [Tempo].[Ano].&[2014] }) ON Columns
from [COMPLETO]
WHERE ({ [Measures].[Faturamento] })
